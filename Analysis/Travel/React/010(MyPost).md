                                                ÏûëÏÑ±Ïûê:{post.userNickname}

                                    </td>

                            ) : (
                                <td>Í≤åÏãúÍ∏ÄÏù¥ ÏóÜÏäµÎãàÎã§.</td>

                        </tr>
                    </tbody>
                </table>

                {/* Í∏ÄÏì∞Í∏∞ Î≤ÑÌäº */}
                <div
                    style={{
                        display: "flex",
                        justifyContent: "flex-end",
                        marginTop: "20px",
                        gap: "20px",//Î≤ÑÌäº Í∞Ñ Í∞ÑÍ≤©

                    <Button
